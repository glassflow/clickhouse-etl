# Phase 3 — Library ↔ Canvas Bridge Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Land the **pinned-vs-live** contract: connections edit live for all pipelines, schemas/transforms are pinned per pipeline revision and only upgrade via an explicit new revision. UI surfaces the contract on the Library Links tab, in-canvas drift banner, per-pipeline upgrade modal, and bulk rollout from Library.

**Architecture:** Adds `pipeline_revisions` and `pipeline_references` tables. Every Deploy from Canvas (Phase 2) writes a `pipeline_revisions` row + a set of `pipeline_references` rows pinning each library resource at the version it was when deployed. Backend `library-links` endpoint joins references against the latest schema/transform versions to compute drift. Upgrade modal creates a new revision with bumped pinned versions. Bulk rollout iterates target pipelines, creates one revision each (atomic = one transaction; staged = sequential with delay).

**Tech Stack:** Drizzle/Postgres · Next.js Route Handlers · Zustand (canvas store re-used) · existing Phase 1 hooks for Library data.

**Branch:** `revamp-2.0/phase-3-bridge`

**Depends on:** Phase 1 + Phase 2 merged.

**Acceptance criteria:** `ACCEPTANCE_CRITERIA.md` § Phase 3.

// src/app/ui-api/pipelines/[id]/revisions/route.ts
import { NextResponse } from 'next/server'
import { desc, eq, max } from 'drizzle-orm'
import { db } from '@/src/lib/db'
import { pipelineRevisions, pipelineReferences } from '@/src/lib/db/schema'
import { CreateRevisionInput } from '@/src/lib/db/validations'

type Params = { params: Promise<{ id: string }> }

export async function GET(_req: Request, { params }: Params): Promise<NextResponse> {
  const { id } = await params
  const rows = await db
    .select()
    .from(pipelineRevisions)
    .where(eq(pipelineRevisions.pipelineId, id))
    .orderBy(desc(pipelineRevisions.revision))
  return NextResponse.json(rows)
}

export async function POST(req: Request, { params }: Params): Promise<NextResponse> {
  const { id } = await params
  const parsed = CreateRevisionInput.safeParse(await req.json())
  if (!parsed.success) {
    return NextResponse.json({ error: parsed.error.flatten() }, { status: 400 })
  }

  // Compute next revision = max(revision) + 1
  const [latest] = await db
    .select({ max: max(pipelineRevisions.revision) })
    .from(pipelineRevisions)
    .where(eq(pipelineRevisions.pipelineId, id))
  const nextRev = (latest?.max ?? 0) + 1

  const result = await db.transaction(async (tx) => {
    const [created] = await tx
      .insert(pipelineRevisions)
      .values({
        pipelineId: id,
        revision: nextRev,
        config: parsed.data.config,
        env: parsed.data.env,
      })
      .returning()

    if (parsed.data.references.length > 0) {
      await tx.insert(pipelineReferences).values(
        parsed.data.references.map((r) => ({
          revisionId: created.id,
          pipelineId: id,
          resourceKind: r.resourceKind,
          resourceId: r.resourceId,
          pinnedVersion: r.pinnedVersion ?? null,
        })),
      )
    }
    return created
  })

  // TODO: kick off backend deploy via existing `deployPipeline` API call
  // For now, the UI-side revision is the source of truth for drift state.

  return NextResponse.json({ pipelineId: id, revision: result.revision, revisionId: result.id }, { status: 201 })
}
```

import { NextResponse } from 'next/server'
import { and, eq } from 'drizzle-orm'
import { db } from '@/src/lib/db'
import { pipelineRevisions, pipelineReferences } from '@/src/lib/db/schema'

type Params = { params: Promise<{ id: string; revision: string }> }

export async function GET(_req: Request, { params }: Params): Promise<NextResponse> {
  const { id, revision } = await params
  const rev = Number(revision)
  if (!Number.isFinite(rev)) {
    return NextResponse.json({ error: 'Invalid revision number' }, { status: 400 })
  }

  const [row] = await db
    .select()
    .from(pipelineRevisions)
    .where(and(eq(pipelineRevisions.pipelineId, id), eq(pipelineRevisions.revision, rev)))
    .limit(1)

  if (!row) return NextResponse.json({ error: 'Not found' }, { status: 404 })

  const refs = await db
    .select()
    .from(pipelineReferences)
    .where(eq(pipelineReferences.revisionId, row.id))

  return NextResponse.json({ ...row, references: refs })
}

- [ ] **Step 1: Implement the route**

```ts
// src/app/ui-api/pipelines/[id]/library-links/route.ts
import { NextResponse } from 'next/server'
import { and, desc, eq, inArray } from 'drizzle-orm'
import { db } from '@/src/lib/db'
import {
  pipelineRevisions,
  pipelineReferences,
  schemas,
  schemaVersions,
  transforms,
  transformVersions,
  kafkaConnections,
  clickhouseConnections,
} from '@/src/lib/db/schema'
import semver from 'semver'

type Params = { params: Promise<{ id: string }> }

type DriftLevel = 'none' | 'patch' | 'minor' | 'major'

type LibraryLink = {
  resourceKind: 'kafka_connection' | 'clickhouse_connection' | 'schema' | 'transform'
  resourceId: string
  resourceName: string | null
  pinnedVersion: string | null   // null = live
  latestVersion: string | null
  drift: DriftLevel
  lastUpgradedAt: string | null
}

export async function GET(_req: Request, { params }: Params): Promise<NextResponse> {
  const { id } = await params

  // Latest revision
  const [latestRev] = await db
    .select()
    .from(pipelineRevisions)
    .where(eq(pipelineRevisions.pipelineId, id))
    .orderBy(desc(pipelineRevisions.revision))
    .limit(1)

  if (!latestRev) return NextResponse.json({ links: [], revision: null })

  const refs = await db
    .select()
    .from(pipelineReferences)
    .where(eq(pipelineReferences.revisionId, latestRev.id))

  if (refs.length === 0) return NextResponse.json({ links: [], revision: latestRev.revision })

  // Resolve names per kind
  const schemaIds = refs.filter((r) => r.resourceKind === 'schema').map((r) => r.resourceId)
  const transformIds = refs.filter((r) => r.resourceKind === 'transform').map((r) => r.resourceId)
  const kafkaIds = refs.filter((r) => r.resourceKind === 'kafka_connection').map((r) => r.resourceId)
  const chIds = refs.filter((r) => r.resourceKind === 'clickhouse_connection').map((r) => r.resourceId)

  const [schemaRows, transformRows, kafkaRows, chRows] = await Promise.all([
    schemaIds.length ? db.select({ id: schemas.id, name: schemas.name }).from(schemas).where(inArray(schemas.id, schemaIds)) : [],
    transformIds.length ? db.select({ id: transforms.id, name: transforms.name }).from(transforms).where(inArray(transforms.id, transformIds)) : [],
    kafkaIds.length ? db.select({ id: kafkaConnections.id, name: kafkaConnections.name }).from(kafkaConnections).where(inArray(kafkaConnections.id, kafkaIds)) : [],
    chIds.length ? db.select({ id: clickhouseConnections.id, name: clickhouseConnections.name }).from(clickhouseConnections).where(inArray(clickhouseConnections.id, chIds)) : [],
  ])

  const nameById = new Map<string, string>()
  for (const r of schemaRows) nameById.set(r.id, r.name)
  for (const r of transformRows) nameById.set(r.id, r.name)
  for (const r of kafkaRows) nameById.set(r.id, r.name)
  for (const r of chRows) nameById.set(r.id, r.name)

  // Latest versions per schema / transform
  const latestSchemaVersions = schemaIds.length
    ? await db
        .select({ schemaId: schemaVersions.schemaId, version: schemaVersions.version, createdAt: schemaVersions.createdAt })
        .from(schemaVersions)
        .where(inArray(schemaVersions.schemaId, schemaIds))
        .orderBy(desc(schemaVersions.createdAt))
    : []
  const latestSchemaByMap = new Map<string, { version: string; createdAt: Date }>()
  for (const v of latestSchemaVersions) {
    if (!latestSchemaByMap.has(v.schemaId)) {
      latestSchemaByMap.set(v.schemaId, { version: v.version, createdAt: v.createdAt })
    }
  }

  const latestTxVersions = transformIds.length
    ? await db
        .select({ transformId: transformVersions.transformId, version: transformVersions.version, createdAt: transformVersions.createdAt })
        .from(transformVersions)
        .where(inArray(transformVersions.transformId, transformIds))
        .orderBy(desc(transformVersions.createdAt))
    : []
  const latestTxByMap = new Map<string, { version: string; createdAt: Date }>()
  for (const v of latestTxVersions) {
    if (!latestTxByMap.has(v.transformId)) {
      latestTxByMap.set(v.transformId, { version: v.version, createdAt: v.createdAt })
    }
  }

  const links: LibraryLink[] = refs.map((r) => {
    let latestVersion: string | null = null
    let lastUpgradedAt: string | null = null
    if (r.resourceKind === 'schema') {
      const latest = latestSchemaByMap.get(r.resourceId)
      latestVersion = latest?.version ?? null
      lastUpgradedAt = latest?.createdAt.toISOString() ?? null
    } else if (r.resourceKind === 'transform') {
      const latest = latestTxByMap.get(r.resourceId)
      latestVersion = latest?.version ?? null
      lastUpgradedAt = latest?.createdAt.toISOString() ?? null
    }

    const drift = computeDrift(r.pinnedVersion, latestVersion)

    return {
      resourceKind: r.resourceKind,
      resourceId: r.resourceId,
      resourceName: nameById.get(r.resourceId) ?? null,
      pinnedVersion: r.pinnedVersion ?? null,
      latestVersion,
      drift,
      lastUpgradedAt,
    }
  })

  return NextResponse.json({ links, revision: latestRev.revision })
}

function computeDrift(pinned: string | null, latest: string | null): DriftLevel {
  if (!pinned || !latest) return 'none'
  const p = semver.coerce(pinned)?.version
  const l = semver.coerce(latest)?.version
  if (!p || !l) return 'none'
  if (semver.eq(p, l)) return 'none'
  if (semver.major(l) > semver.major(p)) return 'major'
  if (semver.minor(l) > semver.minor(p)) return 'minor'
  return 'patch'
}
```

async function findPipelinesByResource(
  kind: 'kafka_connection' | 'clickhouse_connection' | 'schema' | 'transform',
  resourceId: string,

    pipelineId: r.pipelineId,
    pipelineName: r.pipelineId,   // resolve via getPipeline if needed; cheap fallback for now

import useSWR from 'swr'

export type DriftLevel = 'none' | 'patch' | 'minor' | 'major'

export type LibraryLink = {
  resourceKind: 'kafka_connection' | 'clickhouse_connection' | 'schema' | 'transform'
  resourceId: string
  resourceName: string | null
  pinnedVersion: string | null
  latestVersion: string | null
  drift: DriftLevel
  lastUpgradedAt: string | null
}

export type LibraryLinksResponse = { links: LibraryLink[]; revision: number | null }

const fetcher = (url: string) => fetch(url).then((r) => r.json())

export function useLibraryLinks(pipelineId: string | null) {
  const { data, error, isLoading, mutate } = useSWR<LibraryLinksResponse>(
    pipelineId ? `/ui-api/pipelines/${pipelineId}/library-links` : null,
    fetcher,
  )
  const links = data?.links ?? []
  const driftCount = links.filter((l) => l.drift !== 'none').length
  return { links, revision: data?.revision ?? null, driftCount, error: error?.message, isLoading, mutate }
}

type UpgradeModalProps = {

  link: LibraryLink | null
  onClose: () => void
  onUpgraded: () => void

export function UpgradeModal({ pipelineId, link, onClose, onUpgraded }: UpgradeModalProps) {
  const open = link !== null
  const [submitting, setSubmitting] = React.useState(false)

  // Fetch pinned + latest version detail for diff (schema only — transforms shown as code diff in v1)
  const pinnedRes = useSWR(
    open && link?.resourceKind === 'schema' && link.pinnedVersion
      ? `/ui-api/library/schemas/${link.resourceId}/versions/${link.pinnedVersion}`
      : null,
    fetcher,
  )

    open && link?.resourceKind === 'schema' && link.latestVersion
      ? `/ui-api/library/schemas/${link.resourceId}/versions/${link.latestVersion}`
      : null,
    fetcher,
  )

  const handleUpgrade = async () => {
    if (!link) return
    setSubmitting(true)
    try {
      // Fetch current revision config + references, mutate pinnedVersion for this resource, then post a new revision.
      const linksRes = await fetch(`/ui-api/pipelines/${pipelineId}/library-links`).then((r) => r.json())
      const allLinks: LibraryLink[] = linksRes.links ?? []

        resourceKind: l.resourceKind,
        resourceId: l.resourceId,
        pinnedVersion:
          l.resourceKind === link.resourceKind && l.resourceId === link.resourceId
            ? link.latestVersion
            : l.pinnedVersion,
      }))

          references: newReferences,

        notify({ type: 'error', title: 'Upgrade failed' })
        return
      }
      const json = await res.json()
      notify({ type: 'success', title: `Revision ${json.revision} deployed` })
      onUpgraded()
    } finally {
      setSubmitting(false)
    }
  }

// src/app/ui-api/library/schemas/[id]/rollout/route.ts
import { NextResponse } from 'next/server'
import { z } from 'zod'
import { db } from '@/src/lib/db'
import { pipelineRevisions, pipelineReferences } from '@/src/lib/db/schema'
import { and, desc, eq, inArray } from 'drizzle-orm'

const Body = z.object({
  targetPipelineIds: z.array(z.string()).min(1),
  toVersion: z.string().min(1),
  mode: z.enum(['atomic', 'staged']).default('atomic'),
})

type Params = { params: Promise<{ id: string }> }

export async function POST(req: Request, { params }: Params): Promise<NextResponse> {
  const { id: schemaId } = await params
  const parsed = Body.safeParse(await req.json())
  if (!parsed.success) {
    return NextResponse.json({ error: parsed.error.flatten() }, { status: 400 })
  }

  const { targetPipelineIds, toVersion, mode } = parsed.data
  const results: Array<{ pipelineId: string; revision?: number; error?: string }> = []

  for (const pipelineId of targetPipelineIds) {
    try {
      // Fetch latest revision per pipeline
      const [latestRev] = await db
        .select()
        .from(pipelineRevisions)
        .where(eq(pipelineRevisions.pipelineId, pipelineId))
        .orderBy(desc(pipelineRevisions.revision))
        .limit(1)

      if (!latestRev) {
        results.push({ pipelineId, error: 'No revisions' })
        continue
      }

      const refs = await db
        .select()
        .from(pipelineReferences)
        .where(eq(pipelineReferences.revisionId, latestRev.id))

      const newRefs = refs.map((r) => ({
        resourceKind: r.resourceKind,
        resourceId: r.resourceId,
        pinnedVersion:
          r.resourceKind === 'schema' && r.resourceId === schemaId ? toVersion : r.pinnedVersion,
      }))

      const nextRev = latestRev.revision + 1
      const result = await db.transaction(async (tx) => {
        const [created] = await tx
          .insert(pipelineRevisions)
          .values({
            pipelineId,
            revision: nextRev,
            config: latestRev.config,
            env: latestRev.env,
          })
          .returning()
        await tx.insert(pipelineReferences).values(
          newRefs.map((r) => ({
            revisionId: created.id,
            pipelineId,
            resourceKind: r.resourceKind,
            resourceId: r.resourceId,
            pinnedVersion: r.pinnedVersion ?? null,
          })),
        )
        return created
      })

      results.push({ pipelineId, revision: result.revision })

      if (mode === 'staged') {
        await new Promise((r) => setTimeout(r, 500))
      }
    } catch (err) {
      results.push({ pipelineId, error: err instanceof Error ? err.message : 'unknown' })
    }
  }

  const succeeded = results.filter((r) => r.revision != null).length
  return NextResponse.json({ succeeded, total: targetPipelineIds.length, results })
}

Wire `serializeAndDeploy` (Phase 2 task 2.9) to call both: (1) create the revision row in our DB; (2) hit the existing backend deploy endpoint with the same config.
