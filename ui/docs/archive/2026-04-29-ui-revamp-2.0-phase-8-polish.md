This was a known gap in Phase 2. Wire a beforeunload + Next.js router intercept that prompts when `canvasStore.isDirty`.

```tsx
// src/modules/canvas/UnsavedChangesGuard.tsx
'use client'

import * as React from 'react'
import { useStore } from '@/src/store'

export function UnsavedChangesGuard() {
  const { canvasStore } = useStore()

  React.useEffect(() => {
    const handler = (e: BeforeUnloadEvent) => {
      if (!canvasStore.isDirty) return
      e.preventDefault()
      e.returnValue = ''
    }
    window.addEventListener('beforeunload', handler)
    return () => window.removeEventListener('beforeunload', handler)
  }, [canvasStore.isDirty])

  // Note: Next.js App Router doesn't yet expose programmatic interception of <Link> clicks.
  // We rely on beforeunload for full reloads/navigations away from the SPA;
  // for in-app Link clicks, the Phase 2 ConfirmationModal pattern is shown only when
  // user clicks Discard explicitly. This is an accepted simplification.

  return null
}
```

```ts
import type { Node } from '@xyflow/react'

const SOURCE_TYPES = new Set(['kafkaSource', 'otlpSource'])
const SINK_TYPES = new Set(['clickhouseSink'])

export function isValidNodeConnection(source: Node | null | undefined, target: Node | null | undefined): boolean {
  if (!source || !target) return false
  if (source.id === target.id) return false
  if (SINK_TYPES.has(source.type ?? '')) return false
  if (SOURCE_TYPES.has(target.type ?? '')) return false

}
```

```ts
// src/modules/observability/ContextClusterer.ts
import type { LogLine } from '@/src/hooks/useLogsQuery'

export type ClusterItem =
  | { kind: 'line'; line: LogLine }
  | { kind: 'gap'; collapsedCount: number; startIndex: number; endIndex: number }

export function clusterLogs(lines: LogLine[], query: string, contextRadius = 5): ClusterItem[] {
  if (!query.trim()) return lines.map((line) => ({ kind: 'line' as const, line }))

  const term = query.toLowerCase()
  const matchIdx = new Set<number>()
  for (let i = 0; i < lines.length; i++) {
    if (lines[i]._msg.toLowerCase().includes(term)) {
      for (let j = Math.max(0, i - contextRadius); j <= Math.min(lines.length - 1, i + contextRadius); j++) {
        matchIdx.add(j)
      }
    }
  }

  const out: ClusterItem[] = []
  let gapStart: number | null = null

  for (let i = 0; i < lines.length; i++) {
    if (matchIdx.has(i)) {
      if (gapStart !== null) {
        out.push({ kind: 'gap', collapsedCount: i - gapStart, startIndex: gapStart, endIndex: i - 1 })
        gapStart = null
      }
      out.push({ kind: 'line', line: lines[i] })
    } else {
      if (gapStart === null) gapStart = i
    }
  }
  if (gapStart !== null) {
    out.push({ kind: 'gap', collapsedCount: lines.length - gapStart, startIndex: gapStart, endIndex: lines.length - 1 })
  }

  return out
}
```

- [ ] **Step 1: Implement useUrlState**

```ts
// src/hooks/useUrlState.ts
'use client'

import * as React from 'react'
import { usePathname, useRouter, useSearchParams } from 'next/navigation'

export function useUrlState<T extends string>(
  key: string,
  initial: T,
  serialize: (v: T) => string = String,
  deserialize: (s: string) => T = (s) => s as T,
): [T, (v: T) => void] {
  const router = useRouter()
  const pathname = usePathname()
  const params = useSearchParams()

  const value = React.useMemo(() => {
    const v = params.get(key)
    return v != null ? deserialize(v) : initial
    // eslint-disable-next-line react-hooks/exhaustive-deps
  }, [params.get(key)])

  const set = React.useCallback(
    (v: T) => {
      const next = new URLSearchParams(params.toString())
      const s = serialize(v)
      if (s) next.set(key, s)
      else next.delete(key)
      router.replace(`${pathname}?${next.toString()}`, { scroll: false })
    },
    [params, pathname, router, key, serialize],
  )

  return [value, set]
}
```

```tsx
// src/modules/observability/CustomDateRangeModal.tsx
'use client'

import * as React from 'react'
import {
  Dialog, DialogContent, DialogHeader, DialogOverlay, DialogPortal, DialogTitle, DialogFooter,
} from '@/src/components/ui/dialog'
import { Input } from '@/src/components/ui/input'
import { Button } from '@/src/components/ui/button'

type CustomDateRangeModalProps = {
  open: boolean
  initialFrom?: Date | null
  initialTo?: Date | null
  onClose: () => void
  onApply: (range: { fromMs: number; toMs: number }) => void
}

export function CustomDateRangeModal({ open, initialFrom, initialTo, onClose, onApply }: CustomDateRangeModalProps) {
  const [from, setFrom] = React.useState(toLocalIso(initialFrom ?? new Date(Date.now() - 3600_000)))
  const [to, setTo] = React.useState(toLocalIso(initialTo ?? new Date()))

  const handleApply = () => {
    const fromMs = new Date(from).getTime()
    const toMs = new Date(to).getTime()
    if (Number.isFinite(fromMs) && Number.isFinite(toMs) && fromMs < toMs) onApply({ fromMs, toMs })
  }

  return (
    <Dialog open={open} onOpenChange={(o) => !o && onClose()}>
      <DialogPortal>
        <DialogOverlay className="!fixed !inset-0 modal-overlay" aria-hidden="true" />
        <DialogContent className="info-modal-container surface-gradient-border border-0 max-w-md">
          <DialogHeader>
            <DialogTitle className="modal-title">Custom range</DialogTitle>
          </DialogHeader>
          <div className="grid grid-cols-2 gap-3 py-3">
            <div className="flex flex-col gap-1.5">
              <label className="modal-input-label">From</label>
              <Input type="datetime-local" value={from} onChange={(e) => setFrom(e.target.value)} />
            </div>
            <div className="flex flex-col gap-1.5">
              <label className="modal-input-label">To</label>
              <Input type="datetime-local" value={to} onChange={(e) => setTo(e.target.value)} />
            </div>
          </div>
          <DialogFooter>
            <Button variant="ghost" size="sm" onClick={onClose}>Cancel</Button>
            <Button variant="primary" size="sm" onClick={handleApply}>Apply</Button>
          </DialogFooter>
        </DialogContent>
      </DialogPortal>
    </Dialog>
  )
}

function toLocalIso(d: Date): string {
  const off = d.getTimezoneOffset() * 60_000
  return new Date(d.getTime() - off).toISOString().slice(0, 16)
}
```
